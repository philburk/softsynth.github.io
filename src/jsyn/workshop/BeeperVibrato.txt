package workshop;import java.util.*;import java.awt.*;import java.applet.Applet;import com.softsynth.jsyn.*;import JSynExamples.*;/** Add vibrato (frequency modulation) to Beeper*/public class BeeperVibrato extends SynthNote{	public TriangleOscillator  osc;	public SineOscillator      modOsc;	public AddUnit             freqAdd;	/* Add some more ports for vibrato control. */	public SynthInput          modRate;	public SynthInput          modDepth;	/* * Setup synthesis. */	public BeeperVibrato()  throws SynthException	{/* Create various unit generators and add them to circuit. */		add( osc       = new TriangleOscillator() );		add( modOsc    = new SineOscillator() );		add( freqAdd   = new AddUnit() );/* Make ports on internal units appear as ports on circuit. */ 		addPort( amplitude = osc.amplitude );		addPort( frequency = freqAdd.inputB, "frequency" );		addPort( output = osc.output );		/* Modulation rate is the frequency of modulating oscillator. */		addPort( modRate = modOsc.frequency, "modRate" );		addPort( modDepth = modOsc.amplitude, "modDepth" );		/* Connect units together. * Add modulating oscillator output to center frequency * and use result to control oscillator frequency. */		modOsc.output.connect( freqAdd.inputA );		freqAdd.output.connect( osc.frequency );		/* Use frequency signal type so we can set() in Hertz. */		modDepth.setSignalType( osc.frequency );		frequency.setSignalType( osc.frequency );		/* Set preferred range for interactive control, and default value. */		frequency.setup(0.0, 500.0, 3000.0 );		modRate.setup(0.0, 0.5, 40.0 );		modDepth.setup(0.0, 20.0, 1000.0 );		amplitude.setup(0.0, 0.5, 1.0);	}	public void setStage( int time, int stage )  throws SynthException	{		switch( stage )		{		case 0:			start( time );			break;		default:			stop( time );			break;		}	}}
