package workshop;import java.util.*;import java.awt.*;import java.applet.Applet;import com.softsynth.jsyn.*;import com.softsynth.jsyn.circuits.*;import com.softsynth.jsyn.util.*;import com.softsynth.jsyn.view102.*;/** * Play with two parameters using XY controller. *  * @author (C) 1997 Phil Burk, SoftSynth.com, All Rights Reserved */class XYPortController extends XYController{	SynthVariable portX;	SynthVariable portY;	SynthNote     circuit;	/** @param circuit Circuit that will be staged when mouse is pressed or release. Optional, may be null. @param portX port to be controlled by X mouse movement, range set by ports min/max.*/	public XYPortController( SynthNote circuit, SynthVariable portX, SynthVariable portY )	{		this.portX = portX;		this.portY = portY;		this.circuit = circuit;		setMinWorldX( portX.getMin() );		setMaxWorldX( portX.getMax() );		setMinWorldY( portY.getMin() );		setMaxWorldY( portY.getMax() );	}		void useXY(  int x, int y )	{		// System.out.println( "x = " + x + ", y = " + y );		double val;		val = convertGXtoWX( x );		portX.set( val );		val = convertGYtoWY( y );		portY.set( val );	}			/* Respond to 1.0.2 style AWT mouse actions. */	public boolean mouseDown( Event evt, int x, int y )	{		useXY( x, y );	// start note		if( circuit != null ) circuit.setStage( 0 );		return true;	}	public boolean mouseDrag( Event evt, int x, int y )	{		useXY( x, y );		return true;	}	public boolean mouseUp( Event evt, int x, int y )	{		useXY( x, y );	// stop note		if( circuit != null ) circuit.setStage( 1 );		return true;	}}public class XYFun extends Applet{	BeeperVibratoEnv   beeper;	LineOut            lineOut;	XYPortController   xyCtrl;	/* Can be run as either an application or as an applet. */	public static void main(String args[])	{	   XYFun  applet = new XYFun();	   AppletFrame frame = new AppletFrame("Play With It", applet);	   frame.resize(400,460);	   frame.show();	   frame.test();	} /*  * Setup synthesis by overriding Applet's start() method.  */	public void start()  	{		try		{			Synth.startEngine(0);				 // Create circuit and a LineOut so we can hear sound.		 	beeper = new BeeperVibratoEnv()   ;		 	lineOut = new LineOut();		 	beeper.modDepth.set( 200.0 );			// Connect sound to both left and right channels.		 	beeper.output.connect( 0, lineOut.input, 0 );		 	beeper.output.connect( 0, lineOut.input, 1 );						 	lineOut.start();				   } catch(SynthException e) {		  SynthAlert.showError(this,e);	   }	   	   setLayout( new BorderLayout() );	   xyCtrl = new XYPortController( beeper, beeper.modRate, beeper.frequency );	   xyCtrl.setBackground( Color.yellow ); // make box visible	   add( "Center", xyCtrl );	   add( "South", new Label( "Click and drag mouse in box above.", Label.CENTER ) );	   validate();	}	  /*   * Clean up synthesis by overriding Applet's stop() method.   */	public void stop()  	{		removeAll();		lineOut.delete();		Synth.stopEngine();   }}
