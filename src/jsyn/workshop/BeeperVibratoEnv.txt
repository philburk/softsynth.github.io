package workshop;import java.util.*;import java.awt.*;import java.applet.Applet;import com.softsynth.jsyn.*;import JSynExamples.*;/** Add amplitude envelope to BeeperVibrato*/public class BeeperVibratoEnv extends SynthNote{	SawtoothOscillatorBL     osc;	SineOscillator         modOsc;	AddUnit                freqAdd;	SynthEnvelope          envelope;	public EnvelopePlayer  envPlayer;		boolean                lazy = false;	/* Add some more ports for vibrato control. */	public SynthInput          modRate;	public SynthInput          modDepth;	/* * Setup synthesis. */	public BeeperVibratoEnv()  throws SynthException	{/* Create various unit generators and add them to circuit. */		add( osc       = new SawtoothOscillatorBL() );		add( modOsc    = new SineOscillator() );		add( freqAdd   = new AddUnit() );		add( envPlayer = new EnvelopePlayer() );/* Make ports on internal units appear as ports on circuit. */ 		addPort( amplitude = envPlayer.amplitude );		addPort( frequency = freqAdd.inputB, "frequency" );		addPort( output = osc.output );		/* Modulation rate is the frequency of modulating oscillator. */		addPort( modRate = modOsc.frequency, "modRate" );		addPort( modDepth = modOsc.amplitude, "modDepth" );		/* Connect units together. * Add modulating oscillator output to center frequency * and use result to control oscillator frequency. */		modOsc.output.connect( freqAdd.inputA );		freqAdd.output.connect( osc.frequency );		/* Control oscillator amplitude with envelope. */		envPlayer.output.connect( osc.amplitude );/* Create an envelope and fill it with data. */		double[] data =		{			0.05, 1.0,  /* duration,value pair for frame[0] */			0.10, 0.5,  /* duration,value pair for frame[1] */			0.05, 0.7,  /* duration,value pair for frame[2] */			0.05, 0.4,  /* duration,value pair for frame[3] */			0.20, 0.0   /* duration,value pair for frame[4] */		};		envelope = new SynthEnvelope( data );		if( lazy )		{			envelope.setSustainLoop( 2, 4 );		}		/* Set preferred range for interactive control, and default value. */		frequency.setup(0.0, 500.0, 3000.0 );		modRate.setup(0.0, 8.0, 40.0 );		modDepth.setup(0.0, 20.0, 1000.0 );		amplitude.setup(0.0, 0.5, 1.0);	}	public void setStage( int time, int stage )  throws SynthException	{		switch( stage )		{		case 0:			start( time );			if( lazy )			{			/* Queue note ON portion of envelope. */				envPlayer.envelopePort.queueOn( time, envelope );			}			else			{			/* Forget whatever else was in queue. */				envPlayer.envelopePort.clear( time );			/* Queue beginning attack portion. */				envPlayer.envelopePort.queue( time, envelope, 0, 2 );			/* Queue tremolo sustain loop. */				envPlayer.envelopePort.queueLoop( time, envelope, 2, 2 );			}						break;		default:			if( lazy )			{			/* Queue note OFF portion of envelope. */				envPlayer.envelopePort.queueOff( time, envelope );			}			else			{			/* Queue release portion of envelope. */				envPlayer.envelopePort.queue( time, envelope, 4, 1 );			}			break;		}	}}
