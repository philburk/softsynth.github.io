package workshop;import java.util.*;import java.awt.*;import java.applet.Applet;import com.softsynth.jsyn.*;import JSynExamples.*;/** Very simple circuit that only contains a single oscillator.*/public class Beeper extends SynthNote{	public TriangleOscillator  osc;/* * Setup synthesis. */	public Beeper()  throws SynthException	{/* Create various unit generators and add them to circuit. */		add( osc       = new TriangleOscillator() );/* Make ports on internal units appear as ports on circuit. */ 		addPort( amplitude = osc.amplitude );		addPort( frequency = osc.frequency );		addPort( output = osc.output );		/* Set preferred range for interactive control, and default value. */		frequency.setup(0.0, 500.0, 3000.0 );		amplitude.setup(0.0, 0.5, 1.0);	}	public void setStage( int time, int stage )  throws SynthException	{		switch( stage )		{		case 0:			start( time );			break;		default:			stop( time );			break;		}	}}
