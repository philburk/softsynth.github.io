/* Put program in a package to keep classes separate from others. */package workshop;/* Import JSyn classes so we can use them. */import com.softsynth.jsyn.*;/** Simple JSyn program that plays an oscillator with vibrato. * @author Phil Burk (C) 2000 */public class Wiggler{// Declare unit generators that we will use.	public SawtoothOscillatorBL  osc;	public SineOscillator      modOsc;	public AddUnit             freqAdder;	public LineOut             lineOut;// Create some ports to make it easier to change parameters.	SynthInput                 frequency;	SynthInput                 amplitude;	SynthInput                 modRate;	SynthInput                 modDepth;	/** Create unit generators and connect them together. */	public void setupSound()	{	// Create various unit generators.		osc       = new SawtoothOscillatorBL();		modOsc    = new SineOscillator();		freqAdder = new AddUnit();		lineOut   = new LineOut();			// Connect units together.	// Add modulating oscillator output to center frequency	// and use result to control oscillator frequency.		modOsc.output.connect( freqAdder.inputA );		freqAdder.output.connect( osc.frequency );			// Connect oscillator to both left and right channels of output.		osc.output.connect( 0, lineOut.input, 0 );		osc.output.connect( 0, lineOut.input, 1 );	// Make ports with easy to use names.		frequency = freqAdder.inputB;		modRate = modOsc.frequency;		modDepth = modOsc.amplitude;	}/** Start the unit generators so we can hear them. */	public void startSound()	{		osc.start();		modOsc.start();		freqAdder.start();		lineOut.start();	}	/** Wiggle around with various parameters over time. */	public void wiggle( int duration )	{	// Get current time in "ticks".		int time = Synth.getTickCount();			// Set various ports to change sound.		frequency.set( 400.0 );		osc.amplitude.set( 0.5 );		modRate.set( 0.0 );  // turn off modulation		modDepth.set( 0.0 );			// Advance time variable then sleep until that time.		time += duration;		Synth.sleepUntilTick( time );		modRate.set( 8.0 );  // turn on modulation		modDepth.set( 200.0 ); // set in Hz because of setSignalType()			// change frequency		time += duration;		Synth.sleepUntilTick( time );		frequency.set( 800.0 );				time += duration;		Synth.sleepUntilTick( time );	}/** Start JSyn, play some sounds, then stop JSyn. */	public void play()	{		System.out.println("Wiggler - play()");		try		{		// Start JSyn synthesizer.			Synth.startEngine(0);						setupSound();			startSound();			wiggle( 1000 );					// Start JSyn synthesizer.			Synth.stopEngine();					} catch( SynthException e )		{			System.out.println( "Caught " + e );			e.printStackTrace();		}	}/* Main entry point to program. Called by Java Virtual Machine. */	public static void main(String args[])	{	/* This is static method so we need to create an object that we can call. */		Wiggler wiggler = new Wiggler();		wiggler.play();	}}
