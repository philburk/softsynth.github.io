package workshop;import java.util.*;import java.awt.*;import java.applet.Applet;import com.softsynth.jsyn.*;import com.softsynth.jsyn.circuits.*;import com.softsynth.jsyn.util.*;import com.softsynth.jsyn.view102.*;/** Demonstrate how event buffering and absolute sleep solve the timing problems. */public class PlayBeeps extends Applet{	TickerThread        thread1;	int                 advanceTime; // ticks	final static double DEFAULT_ADVANCE = 0.5; // seconds	boolean             go = false;  // control threads	final static int    DUR_MSEC = 400;	int                 duration;	/* Can be run as either an application or as an applet. */	public static void main(String args[])	{/* DO: Change MyJSynProgram to match the name of your class. Must match file name! */	   PlayBeeps  applet = new PlayBeeps();	   AppletFrame frame = new AppletFrame("Play Sequence of Beeps", applet);	   frame.resize(600,120);	   frame.show();	   frame.test();	} /*  * Setup synthesis by overriding Applet's start() method.  */	public void start()  	{		try		{			Synth.startEngine(0);						duration = (int) (Synth.getTickRate() * DUR_MSEC / 1000);						thread1 = new TickerThread( duration, 0.5 );						go = true; // allow threads to run					// start 0.5 seconds in the future			int startTime = Synth.getTickCount() +					(int) (0.5 * Synth.getTickRate());			thread1.start( startTime );				   } catch(SynthException e) {		  SynthAlert.showError(this,e);	   }	}	  /*   * Clean up synthesis by overriding Applet's stop() method.   */	public void stop()  	{	// Set flag so that run() loop will exit the next time around.		go = false;	// wait for thread to finish		try		{			thread1.join( DUR_MSEC + 100 );		} catch( InterruptedException e ) {			System.err.println( e );		}		Synth.stopEngine();   }	/**************************************************************************/class TickerThread extends Thread{	BeeperVibratoEnv   sound;	LineOut            lineOut;	int                startTime;	int                duration;	double             amplitude;		public TickerThread( int duration, double amplitude )  	{		this.duration = duration;		this.amplitude = amplitude;			 // Create circuit and a LineOut so we can hear sound.	 	sound = new BeeperVibratoEnv()   ;	 	lineOut = new LineOut();	// Connect sound to both left and right channels.	 	sound.output.connect( 0, lineOut.input, 0 );	 	sound.output.connect( 0, lineOut.input, 1 );				 	lineOut.start();	}		void start( int startTime )	{		this.startTime = startTime;		this.start();	}		void bang( int time )	{	// calculate random frequency in an octave		double frequency = 200.0 + (Math.random() * 200.0);	// turn on note		sound.noteOn( time, frequency, amplitude );	// turn off note at half duration		sound.noteOff( time + (duration/2) );	}		public void playNotes()	{		advanceTime = (int) (Synth.getTickRate() * DEFAULT_ADVANCE);	// try to start in sync		Synth.sleepUntilTick( startTime  - advanceTime);		int nextTime = startTime;				while( go )  // loop as long as flag is set		{	/* Play a note at the specified time. */			bang( nextTime );	/* Advance nextTime by fixed amount. */			nextTime += duration;	/* sleep until advanceTime BEFORE we have to play the next note */			Synth.sleepUntilTick( nextTime  - advanceTime);		}	}	/* real-time task for thread */	public void run()	{		try		{			playNotes();		}		catch( SynthException e )		{		// Just print message. Do not use Alert because we can easily get an exception		// when we stop the thread.			System.err.println("run() caught " + e );		}	}	}}
